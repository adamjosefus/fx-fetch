import { Tabs, TabItem, Icon } from '@astrojs/starlight/components';

It is easy. You can install FxFetch using your favorite package manager. 


<Tabs syncKey="dependency-manager">
	<TabItem label="npm" icon="seti:npm">
```bash title="Terminal"
npm install fx-fetch
```
	</TabItem>
  
	<TabItem label="pnpm" icon="pnpm">
```bash title="Terminal"
pnpm add fx-fetch
```
	</TabItem>

	<TabItem label="Bun" icon="bun">
```bash title="Terminal"
bun add fx-fetch
```
	</TabItem>

	<TabItem label="Deno" icon="deno">
```bash title="Terminal"
deno install npm:fx-fetch
```
	</TabItem>

	<TabItem label="esm.sh" icon="seti:tsconfig">
```ts showLineNumbers=false title="main.ts"
import { Fetch, Request, Response, Url } from "https://esm.sh/fx-fetch"
```
	</TabItem>
</Tabs>
